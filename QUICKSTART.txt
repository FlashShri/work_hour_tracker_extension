📦 INSTALLATION (3 EASY STEPS)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Open Chrome and type in address bar:
   chrome://extensions/

2. Turn ON "Developer mode" (toggle at top-right)

3. Click "Load unpacked" and select this folder:
   work-hours-tracker/

✅ Done! Look for the clock icon in your toolbar.


🚀 BASIC USAGE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

MORNING (When you arrive):
└─ Click extension icon → Click "🟢 Punch In"

BREAK TIME:
└─ Click "☕ Start Break" → Click "⏸️ End Break" when back

EVENING (When leaving):
└─ Click "🔴 Punch Out"


✨ NEW v2.0 FEATURES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

⚙️  CUSTOM TARGET HOURS:
    └─ Click "⚙️ Settings"
    └─ Change "Target Hours Per Day" (e.g., 7, 8.5, 9)
    └─ Click "💾 Save Settings"

✏️  EDIT PUNCH IN TIME (if you made a mistake):
    └─ Click on the punch in time (shows like "9:00 AM")
    └─ Select correct time
    └─ Click "💾 Update"

📝 MANUAL ENTRY (add past days or custom times):
    └─ Click "✏️ Manual Entry"
    └─ Set punch in time (required)
    └─ Set punch out time (optional for today, required for past)
    └─ Choose date
    └─ Click "💾 Save Entry"


💡 COMMON SCENARIOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📌 Company requires 7.5 hours instead of 8.5:
   → Settings → Change to 7.5 → Save

📌 Forgot to punch in (arrived 9 AM, punched at 10 AM):
   → Punch in now → Click "10:00 AM" → Change to "9:00 AM"

📌 Forgot to track yesterday:
   → Manual Entry → Set times → Change date to yesterday → Save

📌 Working 10 hours today to leave early Friday:
   → Settings → Change to 10 → Work normally → Change back tomorrow


🔔 NOTIFICATIONS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

You'll get notified:
⚠️  At (Target - 30 min) - e.g., 8:00 if target is 8.5
🎉 At Target Hours - e.g., 8.5 hours complete!

The number on the extension icon = hours worked


💡 PRO TIPS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🔧 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Can't edit punch in?
→ Must be currently punched in
→ Use Manual Entry for past days

Manual entry not saving?
→ Past dates need both in AND out times
→ Today needs only in time to start session

Settings not saving?
→ Click "Save Settings" button
→ Refresh the popup
